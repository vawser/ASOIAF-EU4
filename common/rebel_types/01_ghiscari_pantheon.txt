######################################################################
# Worshippers
######################################################################
ghiscari_pantheon_rebels = {

	area = nation 					# Sticks to provinces of their nation
	government = any
	religion = ghiscari_pantheon
	defection = none				# Will not defect
	defect_delay = 120				# Held provinces will auto-defect after 18 months, regardless of accept/enforce demands.
	independence = none				# Does not strive for independence
	unit_transfer = no
	gfx_type = culture_province
	
	resilient = no
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.0
	infantry = 0.9
	cavalry = 0.1
	
	morale = 1.0

	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0.5
			is_overseas = yes
		}
		modifier = {
			factor = 0.0
			owner = { religion = ROOT }
		}
		modifier = {
			factor = 0.0
			NOT = { religion = ghiscari_pantheon }
		}
		modifier = {
			factor = 0.001
			tolerance_to_this = 5
		}
		modifier = {
			factor = 0.5
			tolerance_to_this = 2
		}	
		modifier = {
			factor = 2000
			has_missionary = yes
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = 1 }
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = -1 }
		}
		modifier = {
			factor = 2
			NOT = { tolerance_to_this = -2 }
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.5
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 0.0
			religion = ghiscari_pantheon
			owner = { religion = ghiscari_pantheon }
		}
		modifier = {
			factor = 2.0
			NOT = { religion = ghiscari_pantheon }
			any_neighbor_province = {
	 			religion = ghiscari_pantheon
	 			NOT = { has_province_modifier = "forced_conversion" }
	 		}
	 	}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
 	}
	siege_won_effect = {
		if = {
			limit = {
				religion = ghiscari_pantheon
			}
			add_province_modifier = {
				name = "religious_violence"
				duration = 730
			}
		}
		if = {
			limit = {
				religion_group = religion_ghiscari
			}
			change_religion = ghiscari_pantheon
			add_province_modifier = {
				name = "forced_conversion"
				duration = 730
			}
		}
		if = {
			limit = {
				NOT = { religion_group = religion_ghiscari }
			}
			add_province_modifier = {
				name = "religious_violence"
				duration = 730
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "ghiscari_pantheon_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		add_prestige = -50
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			add_local_autonomy = 20
		}
		if = {
			limit = {
				OR = {
					religion = ghiscari_pantheon
					AND = {
						NOT = { religion = ghiscari_pantheon }
						NOT = { dominant_religion = ghiscari_pantheon }
					}
				}
			}
			add_country_modifier = {
				name = "religious_intolerance"
				duration = 3650
			}
		}
		if = {
			limit = {
				NOT = { religion = ghiscari_pantheon }
				dominant_religion = ghiscari_pantheon
			}
			change_religion = ghiscari_pantheon
			force_converted = yes
		}
	}
}
