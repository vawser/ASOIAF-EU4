white_walker_rebels = {
	area = nation_religion 				# Stays within the nation
	government = any
	defection = none					# Will defect to the "best" alternative.
	independence = none					# Does not strive for independence
	defect_delay = 120
	gfx_type = culture_province
	will_relocate = yes	
	
	resilient = yes					# Disbands on retreat
	reinforcing = yes				# Does not receive reinforcements
	general = yes					# Does not use generals
	smart = yes						# Does not assault forts, tactically withdraw from battle, etc.
	unit_transfer = yes				# Units do not remain after enforcing demands.
	will_relocate = yes				# Units will relocate if they have sieged all provinces nearby and has to move over sea.
	
	artillery = 0.3
	infantry = 0.6
	cavalry = 0.1
	morale = 1.5
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no	
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0
			NOT = { has_global_flag = in_winter }
		}
		modifier = {
			factor = 10.0
			OR = {
				region = region_beyond_the_wall
				region = region_nights_watch
				region = region_the_north
			}
		}
		
		modifier = {
			factor = 0.5
			OR = {
				region = region_dorne
			}
		}
		
		modifier = {
			factor = 0.0
			NOT = { continent = westeros }
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 5
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 2
			development = 6
		}
		modifier = {
			factor = 2
			development = 8
		}
		modifier = {
			factor = 2
			development = 10
		}
		modifier = {
			factor = 2
			development = 12
		}
	}
	
 	# Province Scope
 	siege_won_trigger = {
 		always = yes
 	}
	siege_won_effect = {
        if = { 
            limit = {
                province_id = 1
            }
            cede_province = ---
        }
		if = {
			limit = { 
				is_capital = no
			}
			
			add_base_tax = -2
			add_base_production = -2
			add_base_manpower = -2
		}
		if = {
			limit = { 
				is_capital = yes
			}
			
			add_base_tax = -5
			add_base_production = -5
			add_base_manpower = -5
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = no
	}
	
	# Localisation for their demands
	demands_description = "white_walker_rebels_demand"
	
	# Country scope
	demands_enforced_effect = {
		add_stability = -1
	}
}
