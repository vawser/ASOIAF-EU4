# General Hospital
country_event = {
	id = GOT_DEVOTION.1
	title = GOT_DEVOTION.1.title
	desc = GOT_DEVOTION.1.desc
	picture = WOUNDED_SOLDIERS_eventPicture
	
	trigger = {
		has_dlc = "Common Sense"
		uses_devotion = yes
		NOT = {
			has_country_modifier = dev_general_hospital
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
		
	option = {
		name = GOT_DEVOTION.1.option.a
		ai_chance = {
			factor = 55
			modifier = {
				factor = 0
				has_estate = estate_clergy
				NOT = {
					estate_loyalty = {
						estate = estate_clergy
						loyalty = 40
					}
				}
			}
		}
		if = {
			limit = {
				has_estate = estate_clergy
			}
			add_estate_loyalty = {
				estate = estate_clergy
				loyalty = -10
			}
		}
		add_country_modifier = {
			name = dev_general_hospital
			duration = 3650
		}
	}
	
	option = {
		name = GOT_DEVOTION.1.option.b
		ai_chance = {
			factor = 45
		}
		if = {
			limit = {
				has_estate = estate_clergy
			}
			add_estate_loyalty = {
				estate = estate_clergy
				loyalty = 10
			}
		}
		add_devotion = 10
	}
}
